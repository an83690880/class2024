01. 변수 : 데이터 불러오기

변수는 값을 저장하는 데 사용되며, 이를 통해 프로그램이 데이터를 처리하고 조작할 수 있습니다.
변수는 다양한 유형과 이름을 가질 수 있으며, 값을 변경할 수도 있습니다.
예를 들어, "x"라는 변수에 숫자 5를 할당할 수 있습니다. 그러면 이 변수는 이제 5라는 값을 가지게 됩니다. 이후에 프로그램에서는 "x"를 참조하여 5를 사용하거나 변경할 수 있습니다. 변수는 프로그램의 유연성을 제공하며, 동일한 코드를 사용하여 다양한 데이터를 처리할 수 있도록 도와줍니다. 

{
    num x = 100;
    num y = 200;
    num z = javascript;

    console.log(x); // 콘솔 로그는 프로그래밍에서 디버깅이나 실행 중에 정보를 콘솔(혹은 터미널)에 출력하기 위해 사용되는 기능
    console.log(y); // 콘솔 로그 : 실행문
    console.log(z); 
}
결과 확인하기
100
200
javascript

02. 상수 : 데이터 불러오기

상수는 변수와 비슷한 개념이지만 변수는 값을 여러 번 할당하고 변경할 수 있는 반면, 상수는 한 번 값이 할당되면 그 값을 변경할 수 없습니다.
상수는 변하지 않는 값을 나타내는 데 사용되며, 프로그램 전체에서 고정된 값을 사용해야 할 때 유용합니다. 

{
    const x = 100, y = 200, z = "javascript"; 

    console.log(x, y, z);  
}
결과 확인하기
100
200
javascript

03. 배열 : 데이터 불러오기 : 여러값

배열은 대괄호([]) 안에 쉼표로 구분된 동일한 유형의 데이터 요소들을 순서대로 저장하는 자료구조이고, 여러 유용한 메서드들이 내장되어 있어 요소를 추가, 제거, 검색, 정렬 등 다양한 작업을 수행할 수 있습니다.
각 요소에 대해서는 인덱스를 사용하여 개별적으로 접근할 수 있습니다. 

{
    const num = [100, 200, "javascript"];  

    console.log(num[0], num[1], num[2]); 
}
결과 확인하기
100
200
javascript

04. 배열 : 데이터 불러오기 : 갯수

배열의 length 속성을 활용하여 해당 배열에 저장된 데이터의 개수를 파악할 수 있습니다. 

{
    const num = [100, 200, "javascript"];  

    console.log(num.length);  // 'length' 속성은 배열 내 요소의 개수를 나타냄
}

05. 배열 : 데이터 불러오기 : for()문

for문을 사용하여 배열의 각 요소에 접근하고, 배열의 모든 요소를 반복하여 출력할 수 있습니다. 

{                
    const num = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    for (let i = 0; i < num.length; i++) {               
    console.log(num[i]);
    }
}

06. 배열 : 데이터 불러오기 : forEach()

배열의 forEach() 메서드는 배열의 각 요소에 대해 주어진 함수를 실행하는 메서드입니다.
이 메서드를 사용하면 반복문을 사용하지 않고도 배열의 각 요소를 순회하며 작업을 수행할 수 있습니다. 

{      
    const num = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    //01 각 요소 값 출력
    num.forEach(function (element) {                      
    console.log(element);
    })

    //02 각 요소 인덱스 값 출력
    num.forEach(function (element, index) {
    console.log(index);
    })

    //03 배열 전체 값 출력
    num.forEach(function (element, index, array) {
    console.log(array);
    })          
}

07. 배열 : 데이터 불러오기 : for of

for of 를 통해 배열의 각 요소를 반복하며, 반복할 때마다 현재 요소를 지정된 변수에 할당합니다. for of는 요소의 값을 직접 가져와 반복하기 때문에 코드가 간결하고 가독성이 높아집니다. 

{          
    const num = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    for (let element of num) {   
        console.log(element);   
    }    
}

08. 배열 : 데이터 불러오기 : for in

for in 반복문을 사용해서 배열의 인덱스를 반복하고, 각 인덱스에 해당하는 요소를 참조할 수 있습니다. for in을 배열에 사용할 경우, 배열의 인덱스가 문자열로 처리되며, 배열 이외의 열거 가능한 속성도 순회할 수 있기 때문에 예상치 못한 결과를 초래할 수 있습니다.
따라서 배열을 순회할 때는 주로 for of를 사용하는 것이 좋습니다. 

{          
    const num = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    for (let element in num) {
    console.log(num[element]);  // 배열의 인덱스를 순회하면서, 변수 element에 인덱스를 할당하고, 해당 인덱스의 배열 요소 값을 콘솔에 출력
    }
}

09. 배열 : 데이터 불러오기 : map()

map() 메서드는 배열의 각 요소에 대해 주어진 함수를 호출한 결과를 모아서 새 배열로 반환합니다.
호출된 배열을 변형시키지 않으며, 원본 배열과 동일한 길이의 새 배열을 생성합니다. 

{          
    const num = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    //01 각 요소 값 출력
    num.map(function (element) {
        console.log(element)
    });    

    //02 각 요소 인덱스 값 출력
    num.map(function (element, index) {
        console.log(index)
    });

    //03. 배열 전체 값 출력
    num.map(function (element, index, array) {
        console.log(array)
    });
}
